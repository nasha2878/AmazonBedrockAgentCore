*** system updates ***
sudo apt update

*** Install core utilities  ***
sudo apt install -y \
  curl \
  unzip \
  jq \
  python3 \
  python3-venv \
  ca-certificates

*** Install AWS CLI v2 (ARM64 or x86)  ***
curl "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o awscliv2.zip
                              
                              OR

curl "https://awscli.amazonaws.com/awscli-exe-linux-aarch64.zip" -o awscliv2.zip

unzip awscliv2.zip
sudo ./aws/install
aws --version                          (should be 2.32 or above)


*** python environment ***
python3 -m venv agent-client-env
source agent-client-env/bin/activate
pip install --upgrade pip
pip install boto3

*** CONFIRM AGENTCORE SERVICES ARE AVAILABLE ***
python - <<EOF
import boto3
s = boto3.session.Session()
print("bedrock-agentcore-control" in s.get_available_services())
print("bedrock-agentcore" in s.get_available_services())
EOF
