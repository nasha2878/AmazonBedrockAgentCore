*** PERFORM SYSTEM UPDATES ***
sudo dnf update -y
sudo dnf install -y python3.12 python3-pip
sudo update-alternatives --install /usr/bin/python3 python3 /usr/bin/python3.12 1
python3 --version (should be 3.10 or higher)


*** SETUP A PYTHON VIRTUAL ENVIRONMENT ***
python3 -m venv bedrock-env
source bedrock-env/bin/activate


*** INSTALL REQUIRED PYTHON PACKAGES AND VERIFY THEIR VERSIONS ***
pip install boto3==1.40.49 botocore==1.40.49
pip show boto3 | grep Version


*** INSTALL AWS CLI AND CONFIGURE IT ***
pip install awscli --upgrade
aws configure (using the MyAgentCoreUser user OR your user)


*** INSTALL BEDROCK AGENTCORE STARTER TOOLKIT ***
curl -LsSf https://astral.sh/uv/install.sh | sh 
uv pip install bedrock-agentcore-starter-toolkit
                    OR
pip install bedrock-agentcore-starter-toolkit


*** VERIFY THAT AGENTCORE IS INSTALLED AND CHECK THE VERSION ***
agentcore --help
pip show bedrock-agentcore
