*** System updated ***
sudo apt update && sudo apt upgrade -y

*** Install core utilities  ***
sudo apt install -y \
  ca-certificates \
  curl \
  unzip \
  gnupg \
  lsb-release \
  jq \
  git \
  build-essential

*** Install Python 3.12 and venv ***
sudo apt install -y python3 python3-venv python3-pip    
python3 --version                     (should be above 3.12)
pip3 --version                        (should be above 24.0)


*** Install AWS CLI v2 (ARM64) and CONFIGURE ***
curl "https://awscli.amazonaws.com/awscli-exe-linux-aarch64.zip" -o awscliv2.zip
unzip awscliv2.zip
sudo ./aws/install
aws --version                               (should be above 2.32)


*** Install and Enable Docker (ARM64) ***
curl -fsSL https://get.docker.com | sudo sh
sudo usermod -aG docker $USER
newgrp docker
docker version
docker buildx create --name arm-builder --use
docker buildx inspect --bootstrap             
docker buildx inspect | grep Platforms         (Expected response linux/arm64, linux/amd64)
